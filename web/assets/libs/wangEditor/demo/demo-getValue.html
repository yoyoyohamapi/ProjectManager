wangEditor demo getValue

在demo-basic.html应用的基础上，可以通过 textarea 获取富文本编辑器的源码值。

$('btn1').click(function(e){
	//想要获取编辑框的内容，只需要取得textarea的内容即可
	var html = $('#textarea1').val();
	$('#textareaValue').show();
	$('#textareaValue').val( html );

	//获取编辑框的内容，还有其他两种方式
	//var editor = $('#textarea1').wangEditor();
	//var html = editor.html();  //获取html代码
	//var text = editor.text();  //获取纯文本内容
});
